    	 		\begin{tabular}[h]{|p{0.95\textwidth}|}
			\hline
				Dans la page "Ajouter un bénévole", les champs suivants sont présents : 
				\begin{itemize}

					\item Nom;
					\item Tel fixe;
					\item Tel portable;
					\item Email.
				\end{itemize}
